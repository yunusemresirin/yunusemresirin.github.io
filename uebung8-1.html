8.1 Promise

Content of 'A.txt' and 'B.txt':

Code:

            
    const doc = document.getElementById('content')

    const helper = (function(){
        function getA(){
            return fetch('/A.txt')
                .then((response) => { return response.text() })
        }
        function getB(){
            return fetch('/B.txt')
                .then((response) => { return response.text() })
        }
        return {
            getA,
            getB
        }
    })();

    helper.getA()
    .then((response) => {
        doc.innerText = response;
        return helper.getB()
    })
    .then((response) => {
        doc.innerText += response;
    })
    .catch((error) => {
        doc.innerText = error;
    })
